                          'motorin': ozet.motorin > 0 ? ozet.motorin.toStringAsFixed(2) : '-',

                        return bigBrandKeywords.any((k) => nameLow.contains(k));
                      }).toList();

                        if (m.benzin != null && m.benzin! > 0) {
                          if (minB == null || m.benzin! < minB.benzin!) minB = m;
                          if (maxB == null || m.benzin! > maxB.benzin!) maxB = m;
                        }
                        if (m.motorin != null && m.motorin! > 0) {
                          if (minM == null || m.motorin! < minM.motorin!) minM = m;
                          if (maxM == null || m.motorin! > maxM.motorin!) maxM = m;
                        }
                        if (m.lpg != null && m.lpg! > 0) {
                          if (minL == null || m.lpg! < minL.lpg!) minL = m;
                          if (maxL == null || m.lpg! > maxL.lpg!) maxL = m;
                        }
                      }

                        if (mf == null || fiyat == null || fiyat <= 0) return '-';
                        String name = mf.firma;
                        if (name.toLowerCase().contains('petrol ofisi')) name = 'PO';
                        if (name.toLowerCase().contains('total')) name = 'Total';
                        if (name.toLowerCase().contains('aytemiz')) name = 'Aytemiz';
                        if (name.toLowerCase().contains('shell')) name = 'Shell';
                        if (name.toLowerCase().contains('opet')) name = 'Opet';

                        'name': '🟢 En Ucuz',

                        'motorin': formatCell(minM, minM?.motorin),
                        'lpg': formatCell(minL, minL?.lpg),
                      });
                      
                      tableData.add({
                        'is_header': false,
                        'name': '🔴 En Pahalı',
                        'benzin': formatCell(maxB, maxB?.benzin),
                        'motorin': formatCell(maxM, maxM?.motorin),
                        'lpg': formatCell(maxL, maxL?.lpg),
                      });

                      // Widget'a gönder
